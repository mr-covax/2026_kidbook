#defaultView:Graph
PREFIX gas: <http://www.bigdata.com/rdf/gas#>

# Построение графа связей от "Видеоигры" через иерархию жанров
SELECT DISTINCT ?item1 ?item1Label ?item2 ?item2Label WHERE {
  {
    SELECT ?item1 ?item2 WHERE {
      SERVICE gas:service {
        gas:program gas:gasClass "com.bigdata.rdf.graph.analytics.BFS" ;
                    gas:in wd:Q7889 ;              # стартовая вершина – Видеоигра
                    gas:traversalDirection "Forward" ;
                    gas:out ?item1 ;
                    gas:out1 ?depth1 ;
                    gas:out2 ?item2 ;
                    gas:linkType wdt:P31, wdt:P279 .  # экземпляр и подкласс
      }
    }
  }
  SERVICE wikibase:label { bd:serviceParam wikibase:language "ru,en". }
}
LIMIT 50
